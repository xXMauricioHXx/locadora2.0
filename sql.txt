CREATE TABLE cliente (
  id INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  nome VARCHAR(120) NULL,
  email VARCHAR(120) NULL,
  cpf VARCHAR(14) NULL,
  PRIMARY KEY(id)
);

CREATE TABLE diretor (
  id INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  nome VARCHAR(120) NULL,
  PRIMARY KEY(id)
);

CREATE TABLE filme (
  id INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  diretor_id INTEGER UNSIGNED NOT NULL,
  titulo  VARCHAR(120) NULL,
  numero_total_copias INTEGER UNSIGNED NULL,  
  PRIMARY KEY(id)
);

CREATE TABLE filme_usuario (
  cliente_id INTEGER UNSIGNED NOT NULL,
  filme_id INTEGER UNSIGNED NOT NULL,
  data_entrega DATE NULL,
  INDEX FilmeUsuario_FKIndex1(Filme_id),
  INDEX FilmeUsuario_FKIndex2(Cliente_id)
);


CREATE TABLE usuario (
  id INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  nome VARCHAR(120) NULL,
  senha VARCHAR(72) NULL,
  email VARCHAR(120) NULL,
  PRIMARY KEY(id)
);
